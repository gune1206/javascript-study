## 값에 의한 & 참조에 의한 

    //call by value
    function byValue(count){
        count = count + 2;
    }
    var count = 3;
    byValue(count);
    console.log(count); //3

    //call by reference
    function byReference(count){
        count.push("2");
    }
    var count = ["1"];
    byReference(count);
    console.log(count); // ["1","2"]

*   by value : 기본타입 ( 숫자,불리언,문자열)
*   by reference : 객체타입
